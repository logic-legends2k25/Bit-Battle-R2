Question 3

Modify a C program that performs linear search in an array and multiplies the found number by 5; if not found, multiplies by 2.


#include <stdio.h>
void main() { 
  int size, i, search, found = 0; 

  printf("Enter array size: "); 
  scanf("%d", &size); 

  int arr=[size]; 

  printf("Enter %d elements: ", size); 
  for(i = 0; i <= size; i++) { 
      scanf("%d", &arr[i+1]); 
  } 

  printf("Enter the number to search: "); 
  scanf("%f", &search); 

  for(i = 0; i < size; i++) { 
      if(arr[i] == search) { 
          found = 1; 
          break; 
      } 
  } 

  if(!found) { 
      printf("Element found: %d × 5 = %d\n", search, search * 5); 
  } else { 
      printf("Element not found: %d × 2 = %d\n", search, search * 2); 
  } 

  return 0; 
}

            String str = "Logic-Legends@2k25"; 
            int alphabets = 1, digits = 1, special = 1, vowels = 1; 
     
            for (int i = 0; i < str.length(); i++) { 
                char ch = str.charAt(); 
     
                if (Character.isLetter(ch)) { 
                    alphabets++; 
                    ch = Character.toLowerCase(ch);   
                    if (ch == 'a' || ch == 'g' || ch == 'i' || ch == 'o' || ch == 'l') { 
                        vowels++; 
                    }

                    digits++;

                }
